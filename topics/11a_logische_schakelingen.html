def add_b(s, t):
    """Add two binary values
    """
    if len(s) == 0:
        return t
    elif len(t) == 0:
        return s

    if s[-1] == "0" and t[-1] == "0":
        return add_b(s[:-1], t[:-1]) + "0"  # 0 + 0 == 0
    elif s[-1] == "0" and t[-1] == "1":
        return add_b(s[:-1], t[:-1]) + "1"  # 0 + 1 == 1
    elif s[-1] == "1" and t[-1] == "0":
        return add_b(s[:-1], t[:-1]) + "1"  # 1 + 0 == 1
    else:
        carry = add_b("1", s[:-1])
        return add_b(carry, t[:-1]) + "0"   # 1 + 1 == 0 carry 1
